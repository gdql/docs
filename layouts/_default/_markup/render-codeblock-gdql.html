{{- /* Markdown render hook for ```gdql code blocks. Renders a mac-window-styled code block. Only adds "Try in sandbox" when the block looks like a real runnable query: no square brackets (synopsis), no placeholder words, and starts with a valid GDQL keyword. */ -}} {{- $raw := trim .Inner "\n " -}} {{- $firstLine := index (split $raw "\n") 0 -}} {{- $firstLine = trim $firstLine " " -}} {{- $hasPlaceholders := or (findRE `\[` $raw) (findRE `(?i)\b(condition|sort_spec|date_range|date_or|length_condition|date_spec|venue_or)\b` $raw) -}} {{- $startsWithKeyword := findRE `(?i)^(SHOWS|SONGS|PERFORMANCES|SETLIST|COUNT|FIRST|LAST|RANDOM)\b` $firstLine -}} {{- $isRunnable := and (not $hasPlaceholders) $startsWithKeyword -}} 
{{- $raw -}}
